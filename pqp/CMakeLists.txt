ExternalProject_Add(
	pqp
	#GIT_REPOSITORY https://github.com/GammaUNC/PQP.git
	#GIT_TAG 713de5b70dd1849b915f6412330078a9814e01ab
	URL https://github.com/GammaUNC/PQP/archive/713de5b70dd1849b915f6412330078a9814e01ab.tar.gz
	#URL ${CMAKE_CURRENT_SOURCE_DIR}/PQP-713de5b70dd1849b915f6412330078a9814e01ab.tar.gz
	URL_MD5 8d975c00fd3795344790e68831395ac6
	CMAKE_ARGS
		-DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE}
		-DCMAKE_CFG_INTDIR=${CMAKE_CFG_INTDIR}
		-DCMAKE_INSTALL_PREFIX=${CMAKE_INSTALL_PREFIX}
